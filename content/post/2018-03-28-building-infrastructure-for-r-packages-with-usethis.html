This is a kind of checklist that I use to build consistent package infrastructure with the usethis package.

library(usethis)

Once and for good

use_usethis()

use_description()

use_blank_slate()

Once per package

create_package("PACKAGE-PATH-(ENDING-WITH-PACKAGE-NAME)")

use_rmarkdown_template()
# I use `use_template(template = "CONTRIBUTING.md", package = "fgeo")`

use_tidy_coc()
# Consider moving to .github; use_build_ignore(".github")

use_tidy_contributing()
# I use `use_template(template = "CONTRIBUTING.md", package = "fgeo")`

use_tidy_issue_template()
# I use `use_template(template = "ISSUE_TEMPLATE.md", package = "fgeo")`

use_tidy_support()
# I use `use_template(template = "SUPPORT.md", package = "fgeo")`

# I use `use_template(template = "rmarkdown_template.Rmd", package = "fgeo")`

# I use `use_template(template = "README.Rmd", package = "fgeo")`

Always

use_git()

use_github()

use_github_links()

use_readme_rmd()

use_roxygen_md()

use_news_md()


use_github_labels(delete_default = TRUE)

# *: mit, gpl3, apl2, cc0 -- but see use_description()
use_*_license()

# Lifecycle stages according to https://www.tidyverse.org/lifecycle/
use_lifecycle_badge()

use_cran_badge()

use_package_doc()

Almost always

use_testthat()

# Generally I use `use_coverage("coveralls")`
use_coverage()

use_tidy_ci(browse = interactive())
# Or
use_travis()


use_pipe()

use_pkgdown()

use_tidy_eval()

Continuously, as part of the development process

use_build_ignore()

use_git_ignore()

use_data()

use_data_raw()

# If necessary, set (build)ignore to TRUE.
use_directory("PATH-GOES-HERE", ignore = FALSE)

use_dev_package()

use_dev_version()

use_r()

use_test()

use_vignette()

Ocationally, as part of the development process

use_tidy_description()

use_tidy_versions(overwrite = FALSE)

use_tidy_style()

use_dev_version()

use_version()
