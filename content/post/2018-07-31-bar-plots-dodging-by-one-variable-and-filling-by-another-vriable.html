dataset <- tibble::tribble(
  ~site,  ~sp, ~status,

Create an interaction variable to “dodge manually”.

dodge <- mutate(dataset, site_sp = interaction(site, sp))
dodge

Count.

dodge <- add_count(dodge, site_sp)

Plot.

ggplot(dodge, aes(x = site_sp, y = n, fill = status)) +
  geom_col()
